// src/features/map/layers/ensureExistingCyclingLayer.ts
import maplibregl from "maplibre-gl";

const MARTIN_BASE_URL = import.meta.env.VITE_MARTIN_URL;

// Martin tileset / table name
const CYCLING_SOURCE_ID = "sydney_network_cycling";
const CYCLING_LAYER_ID = "sydney_network_cycling-line";

// Reuse helper: find first symbol layer to insert below labels
const getFirstSymbolLayerId = (map: maplibregl.Map): string | undefined => {
  const style = map.getStyle();
  if (!style?.layers) return undefined;
  for (const layer of style.layers) {
    if (layer.type === "symbol") return layer.id;
  }
  return undefined;
};

export const removeExistingCyclingLayer = (map: maplibregl.Map) => {
  if (map.getLayer(CYCLING_LAYER_ID)) {
    map.removeLayer(CYCLING_LAYER_ID);
  }
  if (map.getSource(CYCLING_SOURCE_ID)) {
    map.removeSource(CYCLING_SOURCE_ID);
  }

  const anyMap = map as any;
  anyMap._existingCyclingEventsBound = false;
};

export const ensureExistingCyclingLayer = (map: maplibregl.Map) => {
  if (!map.isStyleLoaded()) {
    map.once("style.load", () => {
      ensureExistingCyclingLayer(map);
    });
    return;
  }

  const beforeId = getFirstSymbolLayerId(map);
  const anyMap = map as any;

  // 1) Source
  if (!map.getSource(CYCLING_SOURCE_ID)) {
    map.addSource(CYCLING_SOURCE_ID, {
      type: "vector",
      url: `${MARTIN_BASE_URL}/${CYCLING_SOURCE_ID}`,
    } as maplibregl.VectorSourceSpecification);
  }

  // 2) Line layer with color by `facility`
  //
  // Distinct values expected:
  //  - Other bike infrastructure
  //  - Painted bicycle lane
  //  - Shared path
  //  - Separated/protected
  //  - Associated bike infrastructure
  //  - Shared zone/Quietway
  if (!map.getLayer(CYCLING_LAYER_ID)) {
    map.addLayer(
      {
        id: CYCLING_LAYER_ID,
        type: "line",
        source: CYCLING_SOURCE_ID,
        "source-layer": CYCLING_SOURCE_ID,
        paint: {
          "line-color": [
            "match",
            ["get", "facility"],
            "Separated/protected",
            "#22c55e", // green
            "Painted bicycle lane",
            "#3b82f6", // blue
            "Shared path",
            "#eab308", // yellow
            "Shared zone/Quietway",
            "#8b5cf6", // purple
            "Associated bike infrastructure",
            "#ec4899", // pink
            "Other bike infrastructure",
            "#6b7280", // neutral gray
            // default fallback
            "#9ca3af",
          ],
          "line-width": 2.2,
          "line-opacity": 0.9,
        },
        layout: {
          "line-cap": "round",
          "line-join": "round",
        },
      },
      beforeId || undefined
    );
  }

  // 3) Simple interactivity
  if (!anyMap._existingCyclingEventsBound) {
    map.on("mouseenter", CYCLING_LAYER_ID, () => {
      map.getCanvas().style.cursor = "pointer";
    });

    map.on("mouseleave", CYCLING_LAYER_ID, () => {
      map.getCanvas().style.cursor = "";
    });

    map.on("click", CYCLING_LAYER_ID, (e) => {
      const feature = e.features?.[0];
      if (!feature) return;
      console.log("Existing cycling facility:", {
        facility: feature.properties?.facility,
        ...feature.properties,
      });
    });

    anyMap._existingCyclingEventsBound = true;
  }
};
